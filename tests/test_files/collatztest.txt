    cond1: #If even   
        load return_here

        load divarg
        mov D M
	    load div2
        mov 0 ;jmp
	    return_here:

        load divarg

        load test_value

        load con_end

    cond2: #If odd
        load test_value
        mov M D
        add M M
        add M M

    con_end:
        load count

        load test_value

        load heap_address

        mov M A

divcount
divarg

div2: #divide divarg by 2
    load divcount

    divloop:
        load divarg

        div_neg_con:

            load divarg

            add M M

            mov M D
            load 15
            subr A D
